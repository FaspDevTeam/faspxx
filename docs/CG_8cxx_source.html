53FaspRetCode CG::Solve(const VEC& b, VEC& x)
54{
55 FaspRetCode errorCode = FaspRetCode::SUCCESS;
56
57 // Local variables
58 const USI maxStag = MAX_STAG_NUM; // max number of stagnation checks
59 const double solStagTol = 1e-4 * params.relTol; // solution stagnation tolerance
60 const double solZeroTol = CLOSE_ZERO; // solution close to zero tolerance
61
62 USI stagStep = 0, moreStep = 0;
63 double resAbs = 1.0, resRel = 1.0, denAbs = 1.0, ratio = 0.0, resAbsOld = 1.0;
64 double alpha, beta, tmpa, tmpb;
65
66 PrintHead();
67
68 // Initialize iterative method
69 numIter = 0;
70 A->Apply(x, rk); // A * x -> rk
71 rk.XPAY(-1.0, b); // b - rk -> rk
72
73 // Preconditioned search direction
74 zk.SetValues(len, 0.0); // initialize zk = 0
75 pcd->Solve(rk, zk); // preconditioning: B(r_k) -> z_k
76
77 // Prepare for the main loop
78 pk = zk;
79 tmpa = zk.Dot(rk);
80
81 // Main CG loop
82 while (numIter < params.maxIter) {
83
84 // Start checking from minIter instead of 0
85 if (numIter == params.minIter) {
86 resAbs = rk.Norm2();
87 resAbsOld = resAbs; // save initial residual
88 denAbs = (CLOSE_ZERO > resAbs) ? CLOSE_ZERO : resAbs;
89 resRel = resAbs / denAbs;
90 if (resRel < params.relTol || resAbs < params.absTol) break;
91 }
92
93 if (numIter >= params.minIter) PrintInfo(numIter, resRel, resAbs, ratio);
94
95 //---------------------------------------------
96 // CG iteration starts from here
97 //---------------------------------------------
98
99 ++numIter; // iteration count
100
101 A->Apply(pk, ax); // ax = A * p_k, main computational work
102
103 // alpha_k = (z_{k-1}, r_{k-1})/(A*p_{k-1},p_{k-1})
104 tmpb = ax.Dot(pk);
105 if (fabs(tmpb) > CLOSE_ZERO * CLOSE_ZERO)
106 alpha = tmpa / tmpb;
107 else {
108 FASPXX_WARNING("Divided by zero!");
109 errorCode = FaspRetCode::ERROR_DIVIDE_ZERO;
110 break;
111 }
112
113 // Update solution and residual
114 x.AXPY(alpha, pk); // x_k = x_{k-1} + alpha_k*p_{k-1}
115 rk.AXPY(-alpha, ax); // r_k = r_{k-1} - alpha_k*A*p_{k-1}
116
117 //---------------------------------------------
118 // One step of CG iteration ends here
119 //---------------------------------------------
120
121 // Apply several checks for robustness
122 if (numIter >= params.minIter) {
123
124 // Compute norm of residual and output iteration information if needed
125 resAbs = rk.Norm2();
126 resRel = resAbs / denAbs;
127 ratio = resAbs / resAbsOld; // convergence ratio between two steps
128
129 // Save the best solution so far
130 if (numIter >= params.savIter && resAbs < resAbsOld) safe = x;
131
132 // Apply stagnation checks if it converges slowly
133 if (ratio > KSM_CHK_RATIO) {
134 // Check I: if solution is close to zero, return ERROR_SOLVER_SOLSTAG
135 double xNormInf = x.NormInf();
136 if (xNormInf < solZeroTol) {
137 if (params.verbose > PRINT_MIN)
138 FASPXX_WARNING("Iteration stopped due to x vanishes!");
139 errorCode = FaspRetCode::ERROR_SOLVER_SOLSTAG;
140 break;
141 }
142
143 // Check II: if relative difference close to zero, try to restart
144 double xRelDiff = fabs(alpha) * this->pk.Norm2() / x.Norm2();
145 if ((stagStep <= maxStag) && (xRelDiff < solStagTol)) {
146 // Compute and update the residual before restart
147 A->Apply(x, this->rk);
148 this->rk.XPAY(-1.0, b);
149 resAbs = this->rk.Norm2();
150 resRel = resAbs / denAbs;
151 if (params.verbose > PRINT_SOME) {
152 FASPXX_WARNING("Possible iteration stagnate!");
153 WarnRealRes(resRel);
154 }
155
156 if (resRel < params.relTol || resAbs < params.absTol)
157 break;
158 else {
159 if (stagStep >= maxStag) {
160 if (params.verbose > PRINT_MIN)
161 FASPXX_WARNING("Iteration stopped due to stagnation!");
162 errorCode = FaspRetCode::ERROR_SOLVER_STAG;
163 break;
164 }
165 this->pk.SetValues(len, 0.0);
166 ++stagStep;
167 }
168
169 if (params.verbose > PRINT_SOME) {
170 WarnDiffRes(xRelDiff, resRel);
171 FASPXX_WARNING("Iteration restarted due to stagnation!");
172 }
173 } // End of stagnation check!
174
175 } // End of check I and II
176
177 // Check III: prevent false convergence!!!
178 if (resRel < params.relTol) {
179 // Compute and update the true residual r = b - Ax
180 A->Apply(x, this->rk);
181 this->rk.XPAY(-1.0, b);
182
183 // Compute residual norms and check convergence
184 double resRelOld = resRel;
185 resAbs = rk.Norm2();
186 resRel = resAbs / denAbs;
187 if (resRel < params.relTol || resAbs < params.absTol) break;
188
189 // If false converged, print out warning messages
190 if (params.verbose >= PRINT_MORE) {
191 FASPXX_WARNING("False convergence!");
192 WarnCompRes(resRelOld);
193 WarnRealRes(resRel);
194 }
195
196 if (moreStep >= params.restart) {
197 // Note: restart has different meaning here
198 if (params.verbose > PRINT_MIN)
199 FASPXX_WARNING("The tolerance is too small!");
200 errorCode = FaspRetCode::ERROR_SOLVER_TOLSMALL;
201 break;
202 }
203
204 // Prepare for restarting method
205 this->pk.SetValues(0.0);
206 ++moreStep;
207 } // End of check!
208 }
209
210 // Prepare for the next iteration
211 if (numIter < params.maxIter) {
212 // Save the residual for next iteration
213 resAbsOld = resAbs;
214
215 // Apply preconditioner z_k = B(r_k)
216 zk.SetValues(len, 0.0);
217 pcd->Solve(rk, zk);
218
219 // Compute beta_k = (z_k, r_k) / (z_{k-1}, r_{k-1})
220 tmpb = zk.Dot(rk);
221 beta = tmpb / tmpa;
222 tmpa = tmpb;
223
224 // Compute p_k = z_k + beta_k*p_{k-1}
225 pk.XPAY(beta, zk);
226 }
227
228 } // End of main CG loop
229
230 // If minIter == numIter == maxIter (preconditioner only), skip this
231 if (not(numIter == params.minIter && numIter == params.maxIter)) {
232 this->norm2 = resAbs;
233 this->normInf = rk.NormInf();
234 PrintFinal(numIter, resRel, resAbs, ratio);
235 }
236
237 // Restore the saved best iteration if needed
238 if (numIter > params.savIter) x = safe;
239
240 return errorCode;
241}
