53FaspRetCode Jacobi::Solve(const VEC& b, VEC& x)
54{
55 FaspRetCode errorCode = FaspRetCode::SUCCESS;
56
57 // Declaration and definition of local variables
58 double resAbs = 1.0, resRel = 1.0, denAbs = 1.0, ratio = 0.0, resAbsOld = 1.0;
59
60 PrintHead();
61
62 // Initialize iterative method
63 numIter = 0;
64
65 // Main Jacobi loop
66 while (numIter < params.maxIter) {
67
68 // Update residual r = b - A*x
69 A->Residual(b, x, work);
70
71 // Compute norm of residual and check whether it converges
72 if (numIter >= params.minIter) {
73 resAbs = work.Norm2();
74 if (numIter == params.minIter)
75 denAbs = (CLOSE_ZERO > resAbs) ? CLOSE_ZERO : resAbs;
76 resRel = resAbs / denAbs;
77 if (resRel < params.relTol || resAbs < params.absTol) break;
78
79 ratio = resAbs / resAbsOld;
80 resAbsOld = resAbs;
81 PrintInfo(numIter, resRel, resAbs, ratio);
82 }
83
84 //---------------------------------------------
85 // Jacobi iteration starts from here
86 //---------------------------------------------
87
88 ++numIter; // iteration count
89 work.PointwiseMult(diagInv); // r = weight * r ./ d
90 x += work; // x = x + weight * r
91
92 //---------------------------------------------
93 // One step of Jacobi iteration ends here
94 //---------------------------------------------
95
96 } // End of main Jacobi loop
97
98 // If minIter == numIter == maxIter (preconditioner only), skip this
99 if (not(numIter == params.minIter && numIter == params.maxIter)) {
100 A->Residual(b, x, work); // Update final residual
101 this->norm2 = resAbs = work.Norm2();
102 this->normInf = work.NormInf();
103 resRel = resAbs / denAbs;
104 ratio = resAbs / resAbsOld;
105 PrintFinal(numIter, resRel, resAbs, ratio);
106 }
107
108 return errorCode;
109}
