19void MG::oneCycleMultigrid(const VEC &b, VEC &x)
20{
21 // local variables
22 unsigned l = 0; // level index
23 unsigned numCycle[numLevelsUse] = {0};
24
25 // initialize linear system information
26 bVectors[0] = b;
27 xVectors[0] = x;
28
29ForwardSweep:
30 while (l < numLevelsUse - 1) {
31 numCycle[l]++;
32
33 // apply pre-smoothing solver
34 preSolvers[l]->Solve(bVectors[l], xVectors[l]);
35
36 // form residual r = b - A x
37 A->Apply(xVectors[l], wVectors[l]); // SpMV, main computational work
38 bVectors[l].AXPY(-1.0, wVectors[l]); // b = b - A*x
39
40 // restrict residual to coarser level r1 = R*r0
41 restrictions[l]->Apply(bVectors[l], bVectors[l + 1]);
42
43 // prepare for next coarser level
44 ++l;
45 xVectors[l].SetValues(sizes[l], 0.0);
46 }
47
48 // Call the coarsest space solver
49 coarsestSolver->Solve(bVectors[numLevelsUse - 1], xVectors[numLevelsUse - 1]);
50
52 while (l > 0) {
53 --l;
54
55 // prolongate to the next finer level x = x + P*e1
56 prolongations[l]->Apply(xVectors[l + 1], wVectors[l]);
57 xVectors[l].AXPY(1.0, wVectors[l]);
58
59 // apply post-smoothing solver
60 postSolvers[l]->Solve(bVectors[l], xVectors[l]);
61
62 // General cycling on each level
63 if (numCycle[l] < cycles[l])
64 break;
65 else
66 numCycle[l] = 0;
67 }
68
69 if (l > 0) goto ForwardSweep;
70
71 x = xVectors[0];
72}

75FaspRetCode MG::Solve(const VEC &b, VEC &x)
76{
77 FaspRetCode errorCode = FaspRetCode::SUCCESS;
78
79 // Declaration and definition of local variables
80 double resAbs = 1.0, resRel = 1.0, denAbs = 1.0, ratio = 0.0, resAbsOld = 1.0;
81
82 PrintHead();
83
84 // Initialize iterative method
85 numIter = 0;
86
87 // Main Jacobi loop
88 while (numIter < params.maxIter) {
89 // Update residual r = b - A*x
90 A->Residual(b, x, wVectors[0]);
91
92 // Compute norm of residual and check whether it converges
93 if (numIter >= params.minIter) {
94 resAbs = wVectors[0].Norm2();
95 if (numIter == params.minIter)
96 denAbs = (CLOSE_ZERO > resAbs) ? CLOSE_ZERO : resAbs;
97 resRel = resAbs / denAbs;
98 if (resRel < params.relTol || resAbs < params.absTol) break;
99
100 ratio = resAbs / resAbsOld;
101 resAbsOld = resAbs;
102 PrintInfo(numIter, resRel, resAbs, ratio);
103 }
104
105 //---------------------------------------------
106 // Multigrid iteration starts from here
107 //---------------------------------------------
108
109 oneCycleMultigrid(b, x); // MG cycle
110 ++numIter; // iteration count
111
112 //---------------------------------------------
113 // One step of Multigrid iteration ends here
114 //---------------------------------------------
115 } // End of main loop
116
117 // If minIter == numIter == maxIter (preconditioner only), skip this
118 if (not(numIter == params.minIter && numIter == params.maxIter)) {
119 // Update residual r = b - A*x
120 A->Residual(b, x, wVectors[0]);
121 resAbs = wVectors[0].Norm2();
122 resRel = resAbs / denAbs;
123 ratio = resAbs / resAbsOld;
124 this->norm2 = resAbs;
125 this->normInf = wVectors[0].NormInf();
126 PrintFinal(numIter, resRel, resAbs, ratio);
127 }
128
129 return errorCode;
130}
