79template <class TTT>
80void MG<TTT>::MGCycle(const VEC& b, VEC& x)
81{
82 ++level;
83
84 // apply pre-smoothing solver
85 infoHL[level].preSolver->Solve(b, x);
86
87 // skip coarse levels if there is only one level
88 if (numLevelsCoarse > 0) {
89 // form residual r = b - A x
90 A->Residual(b, x, work);
91
92 // restrict residual to coarser level r1 = R*r0
93 infoHL[level].restriction->Apply(work, infoHL[level].b);
94
95 // set coarse initial guess to zero
96 infoHL[level].x.SetValues(infoHL[level].coarSpaceSize, 0.0);
97
98 // call the coarse space solver
99 infoHL[level].coarseSolver->Solve(infoHL[level].b, infoHL[level].x);
100
101 // prolongate the coarse correction back x = x + P*e1
102 infoHL[level].prolongation->Apply(infoHL[level].x, work);
103 x.AXPY(1.0, work);
104 }
105
106 // apply post-smoothing solver
107 infoHL[level].postSolver->Solve(b, x);
108
109 --level;
110}

113template <class TTT>
114FaspRetCode MG<TTT>::Solve(const VEC& b, VEC& x)
115{
116 FaspRetCode errorCode = FaspRetCode::SUCCESS;
117
118 // Declaration and definition of local variables
119 double resAbs = 1.0, resRel = 1.0, denAbs = 1.0, ratio = 0.0, resAbsOld = 1.0;
120
121 PrintHead();
122
123 // Initialize iterative method
124 numIter = 0;
125
126 // Main MG loop
127 while (numIter < params.maxIter) {
128
129 // Update residual r = b - A*x
130 A->Residual(b, x, work);
131
132 // Compute norm of residual and check whether it converges
133 if (numIter >= params.minIter) {
134 resAbs = work.Norm2();
135 if (numIter == params.minIter)
136 denAbs = (CLOSE_ZERO > resAbs) ? CLOSE_ZERO : resAbs;
137 resRel = resAbs / denAbs;
138 if (resRel < params.relTol || resAbs < params.absTol) break;
139
140 ratio = resAbs / resAbsOld;
141 resAbsOld = resAbs;
142 PrintInfo(numIter, resRel, resAbs, ratio);
143 }
144
145 //---------------------------------------------
146 // Multigrid iteration starts from here
147 //---------------------------------------------
148
149 MGCycle(b, x); // MG cycle
150 ++numIter; // iteration count
151
152 //---------------------------------------------
153 // One step of Multigrid iteration ends here
154 //---------------------------------------------
155 } // End of main loop
156
157 // If minIter == numIter == maxIter (preconditioner only), skip this
158 if (not(numIter == params.minIter && numIter == params.maxIter)) {
159 A->Residual(b, x, work); // Update final residual
160 this->norm2 = resAbs = work.Norm2();
161 this->normInf = work.NormInf();
162 resRel = resAbs / denAbs;
163 ratio = resAbs / resAbsOld;
164 PrintFinal(numIter, resRel, resAbs, ratio);
165 }
166
167 return errorCode;
168}
