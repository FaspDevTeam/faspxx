144template <class TTT>
145void MG<TTT>::MGCycle(const VEC& b, VEC& x)
146{
147 if (numLevelsCoarse > 0) {
148
149 ++level;
150
151 // return if out of HL range
152 if (level - numLevelsCoarse == 0) return;
153
154 // pre-smoothing
155 infoHL[level].preSolver->Solve(b, x);
156
157 // form residual r = b - A x
158 A->Residual(b, x, r);
159
160 // restrict residual to coarser level r1 = R*r0
161 infoHL[level].restriction->Apply(r, infoHL[level].b);
162
163 // prepare for coarser level
164 infoHL[level].x.SetValues(infoHL[level].coarSpaceSize, 0.0);
165
166 if (level + 1 - numLevelsCoarse == 0) {
167 // call coarsest space solver
168 infoHL[level].coarseSolver->Solve(infoHL[level].b, infoHL[level].x);
169 } else {
170 // call multigrid cycle recursivley
171 MGCycle(infoHL[level].b, infoHL[level].x);
172 }
173
174 // prolongation P*e1
175 infoHL[level].prolongation->Apply(infoHL[level].x, r);
176
177 // correction x = x + P*e1
178 x.AXPY(1.0, r);
179
180 // post-smoothing
181 infoHL[level].postSolver->Solve(b, x);
182
183 --level;
184
185 } else {
186 // if no HL information, skip MG cycle and return
187 FASPXX_WARNING("No multilevel info available! Do nothing!!!");
188 x = b;
189 }
190}

193template <class TTT>
194FaspRetCode MG<TTT>::Solve(const VEC& b, VEC& x)
195{
196 FaspRetCode errorCode = FaspRetCode::SUCCESS;
197
198 // Declaration and definition of local variables
199 double resAbs = 1.0, resRel = 1.0, denAbs = 1.0, ratio = 0.0, resAbsOld = 1.0;
200
201 PrintHead();
202
203 // Initialize iterative method
204 numIter = 0;
205
206 // Main MG loop
207 while (numIter < params.maxIter) {
208
209 // Update residual r = b - A*x
210 A->Residual(b, x, r);
211
212 // Compute norm of residual and check whether it converges
213 if (numIter >= params.minIter) {
214 resAbs = r.Norm2();
215 if (numIter == params.minIter)
216 denAbs = (CLOSE_ZERO > resAbs) ? CLOSE_ZERO : resAbs;
217 resRel = resAbs / denAbs;
218 if (resRel < params.relTol || resAbs < params.absTol) break;
219
220 ratio = resAbs / resAbsOld;
221 resAbsOld = resAbs;
222 PrintInfo(numIter, resRel, resAbs, ratio);
223 }
224
225 //---------------------------------------------
226 // Multigrid iteration starts from here
227 //---------------------------------------------
228
229 MGCycle(b, x); // MG cycle
230 ++numIter; // iteration count
231
232 //---------------------------------------------
233 // One step of Multigrid iteration ends here
234 //---------------------------------------------
235 } // End of main loop
236
237 // If minIter == numIter == maxIter (preconditioner only), skip this
238 if (not(numIter == params.minIter && numIter == params.maxIter)) {
239 A->Residual(b, x, r); // Update final residual
240 this->norm2 = resAbs = r.Norm2();
241 this->normInf = r.NormInf();
242 resRel = resAbs / denAbs;
243 ratio = resAbs / resAbsOld;
244 PrintFinal(numIter, resRel, resAbs, ratio);
245 }
246
247 return errorCode;
248}
