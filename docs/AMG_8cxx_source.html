40
41 // Step 1. Set problems and temp space for all levels
42 sizes[0] = len;
43 for (unsigned i = 0; i < numLevelsUse; ++i) {
44 bVectors[i].SetValues(sizes[i], 0.0);
45 xVectors[i].SetValues(sizes[i], 0.0);

48

59
