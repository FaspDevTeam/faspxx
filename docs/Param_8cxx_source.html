21static bool ProcessLine(std::string& line, std::string& param, std::string& value)
22{
23 // Remove spaces in the beginning and end of each line
24 line.erase(0, line.find_first_not_of(' '));
25 line.erase(line.find_last_not_of(' ') + 1);
26
27 if (line.empty()) return false; // Skip empty lines
28 if (line[0] == '#') return false; // Skip comments
29 if (line[0] == '%') return false; // Skip comments
30
31 std::string buff;
32 std::stringstream ss(line);
33 ss >> param >> value >> buff;
34 return true;
35}

38static bool JudgeBool(const std::string& value)
39{
40 std::string copy(value);
41 std::transform(value.begin(), value.end(), copy.begin(),
42 [](unsigned char c) { return std::tolower(c); });
43
44 if (copy == "true") {
45 return true;
46 } else if (copy == "false") {
47 return false;
48 } else { // otherwise, an integer
49 return (bool)(std::stoi(copy));
50 }
51}

93void Parameters::ReadFromCommandLine()
94{
95 if (argc == 1) return; // no command line arguments
96
97 for (int i = 1; i < argc; ++i) {
98 // -h, --help will trigger program terminate
99 if (strcmp(argv[i], "-h") == 0 || strcmp(argv[i], "--help") == 0) {
100 PrintHelp();
101 std::exit(FaspRetCode::PRINT_HELP);
102 }
103
104 // check whether the parameters come in pairs: it is still possible the
105 // arguments are in pairs but mismatch!!!
106 if (i >= argc - 1) {
107 std::cout << "Check the argument: " << argv[i] << std::endl;
108 PrintHelp();
109 std::exit(FaspRetCode::PRINT_HELP);
110 }
111
112 // each param is a key-value pair of a Dictionary
113 paramsCML.insert(std::pair<std::string, std::string>(argv[i], argv[i + 1]));
114 ++i;
115 }
116}

119void Parameters::ReadFromFile()
120{
121 for (auto& itm : paramsUser) {
122 if (itm.paramMarker == 2) {
123 std::ifstream file;
124
125 // first try to find param_file from command line params
126 if (paramsCML.find(itm.paramName) != paramsCML.end()) {
127 file.open(paramsCML.find(itm.paramName)->second);
128 } else {
129 file.open(*((std::string*)itm.paramPtr)); // find param_file
130 }
131
132 if (!file.is_open()) FASPXX_WARNING("File not found!");
133
134 std::string line, name, value;
135 while (std::getline(file, line)) {
136 if (ProcessLine(line, name, value)) {
137 // each param is a key-value pair of a Dictionary
138 paramsFile.insert(std::pair<std::string, std::string>(name, value));
139 }
140 }
141 file.close();
142 break; // at most 1 param_file
143 }
144 }
145}

148void Parameters::MergeParams()
149{
150 std::map<std::string, std::string>::iterator iter;
151 for (auto& prm : paramsUser) {
152 iter = paramsFile.find(prm.paramName);
153 if (iter != paramsFile.end()) UpdateParamValue(iter, prm);
154
155 iter = paramsCML.find(prm.paramName);
156 if (iter != paramsCML.end()) UpdateParamValue(iter, prm);
157 }

161void Parameters::UpdateParamValue(std::map<std::string, std::string>::iterator& iter,
162 ParamHolder& prm)
163{
164 switch (prm.paramType) {
165 case BoolType:
166 *(bool*)prm.paramPtr = JudgeBool(iter->second);
167 break;
168 case IntType:
169 *(int*)prm.paramPtr = std::stoi(iter->second);
170 break;
171 case DoubleType:
172 *(double*)prm.paramPtr = std::stod(iter->second);
173 break;
174 case StringType:
175 *(std::string*)prm.paramPtr = iter->second;
176 break;
177 case OutputType:
178 *(Output*)prm.paramPtr = (Output)std::stoi(iter->second);
179 break;
180 default:
181 FASPXX_WARNING("Unknown parameter type!");
182 }
183}
