48{
49 if (params.verbose > PRINT_SOME ||
50 (params.verbose >= PRINT_MIN && iter % PRT_STEP_NUM == 0) ||
51 (params.verbose >= PRINT_MIN && iter == params.maxIter)) {
52 out.precision(4);
53 std::setiosflags(std::ios::scientific);

127void SOL::SetSolTypeFromName(SOLParams& params)
128{
129 for (char& c : params.algName) c = std::tolower(c); // Change to lowercase
130 if (params.algName == "cg")
131 params.type = SOLType::SOLVER_CG;
132 else if (params.algName == "bicgstab")
133 params.type = SOLType::SOLVER_BICGSTAB;
134 else if (params.algName == "minres")
135 params.type = SOLType::SOLVER_MINRES;
136 else if (params.algName == "gmres")
137 params.type = SOLType::SOLVER_GMRES;
138 else if (params.algName == "fgmres")
139 params.type = SOLType::SOLVER_FGMRES;
140 else if (params.algName == "vfgmres")
141 params.type = SOLType::SOLVER_VFGMRES;
142 else if (params.algName == "jacobi")
143 params.type = SOLType::SOLVER_JACOBI;
144 else if (params.algName == "gs")
145 params.type = SOLType::SOLVER_GS;
146 else if (params.algName == "sgs")
147 params.type = SOLType::SOLVER_SGS;
148 else if (params.algName == "sor")
149 params.type = SOLType::SOLVER_SOR;
150 else if (params.algName == "ssor")
151 params.type = SOLType::SOLVER_SSOR;
152 else if (params.algName == "mg")
153 params.type = SOLType::SOLVER_MG;
154 else if (params.algName == "fmg")
155 params.type = SOLType::SOLVER_FMG;
156 else {
157 params.type = SOLType::SOLVER_CG; // default solver type
158 if (params.verbose > PRINT_NONE)
159 FASPXX_WARNING("Unknown solver type. Using default solver!");
160 }
161}
